struct structured_threshold_out_functional : public at::native::structured_threshold_out {
    void set_output_strided(...) override {
        auto current_device = guard_.current_device();
        if (C10_UNLIKELY(current_device.has_value())) {
            TORCH_INTERNAL_ASSERT(*current_device == options.device(),
                "structured kernels don't support multi-device outputs");
        } else {
            guard_.reset_device(options.device());  // 设置 CUDA device
        }
        outputs_[output_idx] = create_out(sizes, strides, options);

1.5 CPU Kernel 实现&注册
实现（cpu/Activation.cpp:153）
void threshold_kernel(
    TensorIteratorBase& iter,
    const Scalar& threshold_scalar,
    const Scalar& value_scalar) {
          AT_DISPATCH_ALL_TYPES(iter.dtype(), "threshold_cpu", [&] {
              using Vec = Vectorized<scalar_t>;
                scalar_t threshold = threshold_scalar.to<scalar_t>();
                Vec threshold_v = Vec(threshold);
                scalar_t value = value_scalar.to<scalar_t>();
                Vec value_v = Vec(value);    
                cpu_kernel_vec(
                    iter,
                    // 标量路径
                    [&](scalar_t x, scalar_t other) -> scalar_t {
                      return x <= threshold ? value : other;
                    },
                    // 向量化路径（AVX2/AVX512）
                    [&](Vec x, Vec other) -> Vec {
                      return Vec::blendv(other, value_v, x <= threshold_v);
                    }
                );
            });
}

static void threshold_kernel_cuda(
    TensorIteratorBase& iter,
    const Scalar& threshold,
    const Scalar& value) {
  AT_DISPATCH_ALL_TYPES_AND2(
      at::ScalarType::Half,
      at::ScalarType::BFloat16,
      iter.dtype(),
      "threshold_cuda",
      [&] {
        threshold_kernel_impl<scalar_t>(
            iter, threshold.to<scalar_t>(), value.to<scalar_t>());
      });
}

template <typename scalar_t>
void threshold_kernel_impl(
    TensorIteratorBase& iter,
    scalar_t threshold,
    scalar_t value) {
  gpu_kernel_with_scalars(
      iter, [=] GPU_LAMBDA(scalar_t x, scalar_t other) -> scalar_t {
        return x <= threshold ? value : other;
      });
}

3. HIP torch.matmal
mm算子上，HIP几乎完全复用了CUDA的实现。
- 差异在 addmm_out_cuda_impl 层的部分Check逻辑
CUDA 代码通过简单的替换（cuda→hip）就能在 AMD 上运行
AMD 强制对齐了 CUDA 的整个 API 设计：
- 流（stream）模型一样
- 内存模型一样
- kernel 启动语法一样
- 函数命名、参数、返回值全一样
NVIDIA GPU 和 AMD GPU 底层架构是同一类：
- SIMD / SIMT 执行模型
- grid/block/thread 三级线程
- 异步流 + 核函数启动。

6. 总结
两个分发机制的对比：
- Dispatch：根据 DispatchKey 选择（算子级）
- Stub：在算子实现内部，根据 DeviceType 选择具体实现（函数级，仍略高于Kernel）
分发方式现状：
- CPU指令集分发：使用 Stub。
- 设备分发：使用 Dispatch + Stub
  - 不同设备有不同的算子Wrapper
  - Stub层收口，CPU不同指令集的、CUDA的实现都会注册到Stub层中
结论：
- 从Stub层注册是可行的
- Wrapper层可能需要改，op.meta() 相关的
- 不是所有算子（如mm）目前的实现都用了Stub，需要先做这层抽象
  - 部分Element-Wise的算子如threshold、cumsum等，已有有stub分发设备的实现
